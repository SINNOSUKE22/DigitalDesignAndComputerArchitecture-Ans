## (b)

各セグメント分のカルノーマップを作るのが大変なので、プログラム(Scheme)で作成する。(多分、どっかでミスする)

``` scheme
;; 真理値表を定義。Grayコード順に並べ替えて、4つ毎に区切っている事に注意。
(define truth-table
  '((("0000" 1 1 1 1 1 1 0)
     ("0100" 0 1 1 0 0 1 1)
     ("1100" X X X X X X X)
     ("1000" 1 1 1 1 1 1 1))
    
    (("0001" 0 1 1 0 0 0 0)
     ("0101" 1 0 1 1 0 1 1)
     ("1101" X X X X X X X)
     ("1001" 1 1 1 0 0 1 1))
    
    (("0011" 1 1 1 1 0 0 1)
     ("0111" 1 1 1 0 0 0 0)
     ("1111" X X X X X X X)
     ("1011" X X X X X X X))
    
    (("0010" 1 1 0 1 1 0 1)
     ("0110" 1 0 1 1 1 1 1)
     ("1110" X X X X X X X)
     ("1010" X X X X X X X))))

;; 真理値表をカルノーマップの並びに変換
(define karnaugh-maps 
  (map (lambda (index)
	 (map (lambda (row)
		(map (lambda (col)
		       (list-ref col index))
		     row))
	      truth-table))
       '(1 2 3 4 5 6 7)))

;; 各セグメント毎にカルノーマップを出力
(for-each (lambda (seg-name a-map)
	    (newline)
	    (display seg-name)
	    (newline)
	    (display "    | 00 | 01 | 11 | 10")
	    (newline)
	    (display "----+----+----+----+----")
	    (newline)
	    (for-each (lambda (head low)
			(display head)
			(for-each (lambda (col)
				    (format #t "|  ~A " col))
				  low)
			(newline))
		      '(" 00 " " 01 " " 11 " " 10 ")
		      a-map))
	  '("S-a" "S-b" "S-c" "S-d" "S-e" "S-f" "S-g")
	  karnaugh-maps)
```
